# Introduction

Webhooks allow your application to be notified in real time when events occur.
This guide shows how to receive and verify **Deposit**, **Convert**, and **Transfer** webhook events.


# Receiving an Event

You need to expose an endpoint in your server to receive webhook events.
Below are examples in **TypeScript**, **Go**, and **Python**.
<CodeGroup>

```ts TypeScript
import express from "express";

const app = express();
app.use(express.json());

app.post("/webhook", (req, res) => {
  const signature = req.headers["x-webhook-signature"] as string;
  const payload = req.body;

  console.log("Received event:", payload);
  res.sendStatus(200);
});

app.listen(3000, () => console.log("Webhook server running on port 3000"));
```

```go Golang
package main

import (
	"encoding/json"
	"fmt"
	"net/http"
)

func webhookHandler(w http.ResponseWriter, r *http.Request) {
	var payload map[string]interface{}
	if err := json.NewDecoder(r.Body).Decode(&payload); err != nil {
		http.Error(w, "invalid payload", http.StatusBadRequest)
		return
	}

	signature := r.Header.Get("X-Webhook-Signature")
	fmt.Println("Received signature:", signature)
	fmt.Println("Received event:", payload)

	w.WriteHeader(http.StatusOK)
}

func main() {
	http.HandleFunc("/webhook", webhookHandler)
	fmt.Println("Server running on :3000")
	http.ListenAndServe(":3000", nil)
}
```

```python Python
from fastapi import FastAPI, Request

app = FastAPI()

@app.post("/webhook")
async def webhook(request: Request):
    payload = await request.json()
    signature = request.headers.get("x-webhook-signature")

    print("Received signature:", signature)
    print("Received event:", payload)

    return {"status": "ok"}
```

# Verifying Event Source

Each webhook includes a `signature` that you must verify using your configured public key.
Below is a verification **handler pattern** in **TypeScript**, **Go**, and **Python**.

<CodeGroup>

```ts TypeScript
import crypto from "crypto";

const publicKey = `-----BEGIN PUBLIC KEY-----
MIIBIjANBgkqhkiG9w0BAQEFAAOCAQ8AMIIBCgKCAQEAtVdummyPublicKeyHere
-----END PUBLIC KEY-----`;

export function verifySignature(payload: string, signature: string): boolean {
  const verifier = crypto.createVerify("RSA-SHA256");
  verifier.update(payload);
  verifier.end();

  return verifier.verify(publicKey, Buffer.from(signature, "base64"));
}
```

```go Golang
package main

import (
	"crypto"
	"crypto/rsa"
	"crypto/sha256"
	"crypto/x509"
	"encoding/base64"
	"encoding/pem"
)

var publicKeyPEM = []byte(`-----BEGIN PUBLIC KEY-----
MIIBIjANBgkqhkiG9w0BAQEFAAOCAQ8AMIIBCgKCAQEAtVdummyPublicKeyHere
-----END PUBLIC KEY-----`)

func verifySignature(payload []byte, signature string) bool {
	block, _ := pem.Decode(publicKeyPEM)
	if block == nil {
		return false
	}
	pub, err := x509.ParsePKIXPublicKey(block.Bytes)
	if err != nil {
		return false
	}
	rsaPub := pub.(*rsa.PublicKey)

	hashed := sha256.Sum256(payload)
	sigBytes, _ := base64.StdEncoding.DecodeString(signature)

	err = rsa.VerifyPKCS1v15(rsaPub, crypto.SHA256, hashed[:], sigBytes)
	return err == nil
}
```

```python Python
import base64
from cryptography.hazmat.primitives import hashes
from cryptography.hazmat.primitives.asymmetric import padding
from cryptography.hazmat.primitives import serialization

public_key_pem = b"""-----BEGIN PUBLIC KEY-----
MIIBIjANBgkqhkiG9w0BAQEFAAOCAQ8AMIIBCgKCAQEAtVdummyPublicKeyHere
-----END PUBLIC KEY-----"""

public_key = serialization.load_pem_public_key(public_key_pem)

def verify_signature(payload: str, signature: str) -> bool:
    try:
        public_key.verify(
            base64.b64decode(signature),
            payload.encode(),
            padding.PKCS1v15(),
            hashes.SHA256()
        )
        return True
    except Exception:
        return False
```
